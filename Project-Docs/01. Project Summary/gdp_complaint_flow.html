flowchart LR
    START([Complaint email arrives:
Customer Service / Supply Chain]) --> INTAKE[Intake Agent:
extract customer, product, batch/lot, qty,
order/invoice ref, description, attachments]
    INTAKE --> ERPPULL["ERP: pull Sales Order for comparison
(robot reads ERP desktop app)"]
    ERPPULL --> CLASSIFY{Classify family
and severity}
    CLASSIFY --> G1{{"GATE 1 — Triage validation
Owner: Customer Service
Valid, well-classified complaint?"}}
    G1 -->|spam / misroute| REJECT([Close: not a complaint])
    G1 -->|Valid| ACK[Auto-acknowledge customer
create case in Cockpit + open audit trail]

    ACK --> COMPLETE{Enough info
to investigate?}
    COMPLETE -->|No| RFI[RFI to customer
park on SLA timer]
    RFI -.->|reply| COMPLETE
    RFI -.->|SLA breach| ESC[Escalate / chase] -.-> COMPLETE

    COMPLETE -->|Yes| CORR{Cross-case correlation:
same dispatch/batch +
opposite qty delta?}
    CORR -->|Match found| LINK[Link cases
X short and Y surplus
resolve together]
    CORR -->|No match| INV
    LINK --> INV

    INV{Route by family} -->|A: Quality defect| INVA[Pull batch record, prior complaints,
QC release, photos, same-batch spread]
    INV -->|B: Cold-chain| INVB[Pull logger data, stability/excursion budget,
lane, carrier records]
    INV -->|C: Logistics / qty error| INVC["CS/Supply Chain VALIDATION:
warehouse inventory check +
ERP sales-order vs shipped qty"]

    INVA --> DISPQ
    INVB --> DISPQ
    INVC --> CLAIMVALID{Claim valid?
and goods return required?}

    CLAIMVALID -->|"Valid, NO goods return
(pure qty / human error)"| COMM["COMMERCIAL LANE
Supply Chain decides:
credit / debit / redeliver"]
    CLAIMVALID -->|Invalid / unfounded| NORET[Close, no action]
    COMM --> QANOTIFY[/"QA NOTIFY (non-blocking)
metrics + trending only"/]
    QANOTIFY --> G5

    CLAIMVALID -->|"Valid, goods WILL return"| DISPQ
    DISPQ[Case Assembly Agent:
structured dossier + GDP 6.3 criteria
TRUE/FALSE/UNKNOWN + recommendation]
    DISPQ --> ROOT[/"ROOT CAUSE entered
(human error, transit damage,
cold-chain fail, pick error)
mandatory before QA"/]
    ROOT --> G2{{"GATE 2 — Investigation sign-off
Owner: QA Analyst
Evidence complete?"}}
    G2 -->|Insufficient / UNKNOWN| BACK[Loop: more RFI / evidence] -.-> COMPLETE

    G2 -->|Complete| POLICY{Policy router:
product + type matches
standing disposition policy?}

    POLICY -->|"PATH A: policy match"| FORK([FORK: Authority + Execution tracks])
    FORK --> G3A{{"GATE 3 — RP Ratification
Owner: Responsible Person
confirm policy applied"}}
    FORK --> PREP[Execution PREPARATORY only
nothing irreversible]
    PREP --> G4P{{"GATE 4 — Plan approval
Owner: Logistics"}}
    PREP --> G5P{{"GATE 5 — Finance DRAFT
Owner: Finance"}}
    G4P --> PREPDONE[pickup scheduled / stock reserved
paperwork drafted]
    G5P --> CREDITDRAFT[credit/debit DRAFTED
not posted]

    G3A --> JOIN{JOIN:
RP x execution state}
    PREPDONE --> JOIN
    CREDITDRAFT --> JOIN
    JOIN -->|Ratified| COMMIT[COMMIT: post finance to ERP,
dispatch redelivery]
    JOIN -->|"Override, nothing committed"| OV1[cancel plan, void DRAFT] --> G3B
    JOIN -->|"Override, credit posted"| OV2[Saga reversal:
reversing note, reconcile ERP] --> DEV[Deviation / CAPA] --> G3B
    COMMIT --> RETURNFLOW

    POLICY -->|"PATH B: no match"| G3B{{"GATE 3 — Disposition (GDP)
Owner: RP / QA
back-to-stock vs destroy
per 6.3 criteria"}}
    G3B --> RETURNFLOW

    RETURNFLOW["Goods return to warehouse (robot in ERP):
adjust batch quantity
set status = QUARANTINE"]
    RETURNFLOW --> G3D{{"GATE 3 — QA disposition on quarantined batch
Owner: RP / QA
(root cause + 6.3 reviewed)"}}
    G3D -->|All 6.3 TRUE - authorize| AVAIL["ERP: batch status = AVAILABLE
FEFO placement"]
    G3D -->|any FALSE/UNKNOWN| BLOCK["ERP: batch status = BLOCKED /
AWAITING DESTRUCTION"]
    BLOCK --> DESTROY["DESTRUCTION — irreversible,
only after RP authorization
- certificate of destruction"]

    AVAIL --> G4B
    DESTROY --> G4B
    G4B{{"GATE 4 — Follow-up plan
Owner: Logistics"}} --> EXECB[arrange return / redelivery]
    EXECB --> G5
    G5{{"GATE 5 — Finance approval
Owner: Finance
approve credit / debit amount"}} --> FIN["ERP: post credit / debit note
(robot drives ERP desktop app)"]

    FIN --> CLOSE
    NORET --> CLOSE
    CLOSE([Notify customer, close case,
QA metrics updated, full audit trail])
